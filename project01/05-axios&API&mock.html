    # 项目一前台API代码

    # 无参

    请求地址：api/product/getBaseCategoryList 
请求方式：get 
请求参数：无参数

export const reqCategoryList = () =>
  requests({ url: "/product/getBaseCategoryList", method: "get" });

    # 一个对象参数

    请求地址：/api/list
请求方式：post    
请求参数：需要带参数，类型为对象，放到请求的data属性中，且至少要是一个空对象，不能是null，否则服务器返回数据会报错

export const reqGetSearchInfo = (data) =>
  requests({ url: "/list", method: "post", data});

    # 一个普通参数

    请求地址：/api/item/{skuId}
请求方式：get    
请求参数：skuId

export const reqGetGoodInfo = (skuId) =>
  requests({ url: `/item/${skuId}`, method: "get" });

    # 两个普通参数

    请求地址：/api/cart/addToCart/{skuId}/{skuNum}
请求方式：post    
请求参数：skuId skuNum

export const reqAddOrUpdateCart = (skuId, skuNum) =>
  requests({ url: `/cart/addToCart/${skuId}/${skuNum}`, method: "post" });

    # 项目一封装axios & 设置请求相应拦截器

    1、利用axios对象的create方法，创建一个axios实例request，并加上基础路径和请求超时时间的设置；

    2、请求拦截器：在发请求之前，将其拦截，在请求的headers请求头中加入uuid和token以作为用户的唯一凭证，再发出去；

    3、响应拦截器：将服务器返回的请求拦截，并将返回结果中的data属性作为返回值；所以才能，在发请求之后，才能异步获取返回的结果。

    import axios from "axios";
import store from "@/store";
// axios
const requests = axios.create({
  baseURL: "/api",		// 设置基础路径，发请求的时候，路径当中会出现api
  timeout: 5000,		// 请求超时的时间为5S
});

// 请求拦截器
requests.interceptors.request.use((config) => {
  // config：配置对象，对象里面有一个属性很重要，headers请求头
  if (store.state.detail.uuid_token) {
    // 在每次的网络请求的请求头中加入userTempId(项目后台规定的变量)，其内容是项目前台生成的本机游客的唯一凭证uuid，
    config.headers.userTempId = store.state.detail.uuid_token;
  }
  if (store.state.user.token) {
    // 在每次的网络请求的请求头中加入token(项目后台规定的变量)，其内容是服务器为本用户生成的唯一凭证token，
    config.headers.token = store.state.user.token;
  }
  return config;
});

// 响应拦截器
requests.interceptors.response.use(
  (res) => {
    // 成功的回调函数：服务器相应数据回来以后，响应拦截器可以检测到，可以做一些事情
    return res.data;
  },
  (error) => {
    // 响应失败的回调函数
    return Promise.reject(error);
  }
);
// let result = await reqAddOrUpdateCart(skuId, skuNum); 	// 调用api里的接口函数reqGetGoodInfo；
// 因为上面的响应拦截器把返回结果的data作为return，所以result才能等于res.data
// 服务器返回示例：
// {
//   code: 200
//   message: "成功"
//   data:{
//       nickName:"Administrator",
//       name:"Admin",
//       token: 90aa16f24d04c7d882051412f9ec45b"
//   }
//   ok: true 
// }
// {
//   code: 201
//   message: "失败"
//   data: null
//   ok: false 
//  }

export default requests;

    # 跨域问题

    当一个请求url的协议,域名,端口三者之间任意一个与当前的url不同都即为跨域。项目一使用代理服务器解决。

    http://localhost:8080/#/home	----前端本地
http://39.98.123.211			----后台服务器

项目的请求网址为：http://localhost:8080/api/order/auth/submitOrder?tradeNo，而本地没有对于api的响应接口，所以，
在vue.config.js中加入如下代码
module.exports = defineConfig({
  devServer: {
    proxy: { "/api": { target: "http://gmall-h5-api.atguigu.cn" } },
  },
});
// 碰到url中带有"/api"的请求，向target指定的服务器获取数据。

    # mock模拟数据

    mock是接口测试工具，在不实际启动后端服务器的情况下，通过模拟请求发送到挡板（mock服务），挡板接收到请求，返回程序员提前模拟的服务器返回的数据。

    # mock配置

    1、安装
npm install mockjs@1.1.0
2、src目录下新建mock文件夹，文件夹内创建需要的JSON文件，里面存储模拟返回的JSON数据
比如，banner.json文件里的数据就模拟的服务器返回的banner数据：
[
  {
    "id": "1",
    "imgUrl": "/images/banner1.jpg"
  },
  {
    "id": "2",
    "imgUrl": "/images/banner2.jpg"
  },
  {
    "id": "3",
    "imgUrl": "/images/banner3.jpg"
  },
  {
    "id": "4",
    "imgUrl": "/images/banner4.jpg"
  }
]
3、mock数据中图片等资源的注意
把mock数据需要的图片放置到public文件夹中【public文件夹在打包的时候，会把相应的资源原封不动打包到dist文件夹中】，与资源路径保持一致

    4、mock文件夹下，创建mockServe.js文件
// 引入mockjs模块
import Mock from "mockjs";
// 引入JSON数据
// webpack 默认对外暴露：图片、JSON数据，所以不需要额外加入export语句
import banner from "./banner.json";
import floor from "./floor.json";

// 下面的Mock是mockjs对外暴露的引用对象，mock是其中的一个方法，需要两个参数，参数1是请求地址，参数2是请求返回的数据；表示当有请求地址为参数1的地址时，返回参数2的模拟数据
Mock.mock("mock/banner", { code: 200, data: banner });
Mock.mock("mock/floor", { code: 200, data: floor });

    5、在main.js文件中引入mockServe文件，目的是至少让该文件执行一次，才能使用其中的方法
import "@/mock/mockServe"; // 引入mockServe.js文件

    # 设置发送请求的配置

    给请求加上mock路径，代表此请求是向mock发出的。

    1、api文件夹内新建mockAjax.js文件

import axios from "axios";
const requests = axios.create({
  baseURL: "/mock",
  timeout: 5000,
});

// 请求拦截器：mock也能在发请求之前，通过请求拦截器给config中的headers请求头加数据
requests.interceptors.request.use((config) => {
  return config;
});

// 响应拦截器
requests.interceptors.response.use(
  (res) => {
    return res.data;
  },
  (error) => {
    return Promise.reject(error);
  }
);
export default requests;

    2、api文件夹内的index.js文件
import mockRequests from "./mockAjax";
// 获取banner数据
export const reqGetBannerList = () => mockRequests({ url: "/banner", method: "get" });

注意：mockServe需要的请求路径由以下组成：mockAjax.js文件里的基础路径，和此处的url
