    介绍

    # 介绍

    canvas是HTML5的新标签，是一个图形容器，通过JavaScript绘制图像。绘制出的图形不是DOM，而是像素点的展示，在网页上的DOM节点仍然是只有canvas。

    # 初始化

    HTML标签，初始化画布容器。

    <canvas width="800" height="400"></canvas>		// 设置画布宽高，最好只用这种方式设置宽高
// <canvas><h1>好好学习</h1></canvas>	// 给canvas标签添加子节点、添加文本，没有意义。

    JavaScript，初始化画笔

    let canvas = document.querySelector('canvas');
let ctx = canvas.getContext('2d');
// 获取到canvas画布所在的节点
// ctx 是一个对象，里面有很多属性方法，用来绘制图形

    # 图形绘制

    # 线段

    设置起点：ctx.moveTo(100,100);	// (0,0)在画布的左上角
设置终点：ctx.lineTo(200,200);	// lintTo后，该终点成了新起点
再次设置终点：ctx.lineTo(200,100);
画出线段：ctx.stroke();

    # 矩形

    ctx.fillRect(x,y,width,height);		//绘制填充颜色的矩形
ctx.strokeRect(x,y,width,height);	//绘制线条的矩形
// (x,y)矩形左上角的坐标，width，height矩形宽高

    # 圆/扇形/圆弧

    ctx.arc(x, y, radiusradius, starAngle, endAngle, anticlockwise)
// (x,y)圆心坐标、radius圆半径、anticlockwise(true逆时针，false顺时针)
// starAngle开始角度, endAngle结束角度。（角度计算：圆的最右点为0，1*Math.PI是180度）
// 总之，调整开始/结束角度，调整Math.PI的乘数的大小正负，顺逆时针，可以得到不同的扇形/圆弧
圆：
ctx.arc(100, 100, 30, 0, 2*Math.PI, false);
ctx.fill();
扇形：
ctx.arc(100, 100, 30, 0.2*Math.PI, 1.2*Math.PI, false);
ctx.fill();
圆弧：
ctx.arc(100, 100, 30, -0.2*Math.PI, 1.2*Math.PI, false);
ctx.stroke();

    # 文本

    ctx.font="20px 微软雅黑";
ctx.fillText("数据可视化",x,y);
// (x,y)是文本左上角的位置坐标

    # 样式

    ctx.fillStyle = "red";		// 闭合图形的填充色
ctx.strokeStyle = "red";	// 线条的颜色
ctx.lineWidth = 5;			// 线条宽度
// 设置样式后，后面的图形绘制都会使用该样式属性，直到再遇到新的样式语句代替它；

    # 清除

    重置绘制数据，清除前面的图形绘制：ctx.beginPath();
清除画布中的矩形区域：ctx.clearRect(x,y,width,height)，参数与绘制矩形的参数一致
设置起止点连接：ctx.closePath();		// ctx.stroke()画出线段前调用此方法，会将最后一次的起点和终点连接起来。

    # 坐标系

    移动原点：
// translate(x,y)：translate接受两个值，x表示向左右的偏移量，正数代表向右偏移，负数向左；y代表上下的偏移量，正数向下，负数向上。
ctx.translate(0, height);		// 原点移动到画布左下角

翻转坐标系：
ctx.scale(1, -1);		
// scale原本是缩放画布函数。scale(x,y)：x缩放当前绘图的宽度（1=100%，0.5=50%，2=200%，负数代表向原点反向绘制，依次类推）。y是高度。
// scale(1,-1)，则是横向不变，纵向反向绘制，相当于将原坐标系上下翻转。

    # 一个柱形图的绘制

    <canvas id="myCanvas" width="750" height="500" style="border:1px solid #d3d3d3;"></canvas>

    // 初始化画笔
let c = document.getElementById("myCanvas");
let ctx = c.getContext("2d");

// 将原点设置到画布左下角，坐标系转换为常用的右上坐标系
ctx.translate(50, 450);
// 绘制文本
ctx.font = "20px 微软雅黑";
ctx.fillText("FOOD", 50, 30);
ctx.fillText("DIGITAL", 205, 30);
ctx.fillText("LUGGAGE", 355, 30);
ctx.fillText("FASHION", 515, 30);
ctx.fillText("30", -40, -90);
ctx.fillText("60", -40, -190);
ctx.fillText("90", -40, -290);
ctx.fillText("120", -50, -390);
// 将图形的绘制上下翻转；（文本在翻转之前绘制是因为，图形被翻转绘制不影响外观，但是文本被翻转就会无法识别）
ctx.scale(1, -1);

// 画坐标线
ctx.fillStyle = "#131124";
ctx.lineWidth = 2;
ctx.moveTo(-10, 0);
ctx.lineTo(640, 0);
ctx.moveTo(0, -10);
ctx.lineTo(0, 420);
for (let i = 1; i < 4; i++) {
    ctx.moveTo(160 * i, 0);
    ctx.lineTo(160 * i, -10);
}
for (let i = 1; i < 5; i++) {
    ctx.moveTo(0, 100 * i);
    ctx.lineTo(-10, 100 * i);
}
ctx.fillStyle = "#ccccd6";
ctx.lineWidth = 0.2;
for (let i = 1; i < 5; i++) {
    ctx.moveTo(0, 100 * i);
    ctx.lineTo(800, 100 * i);
}
ctx.stroke();

// 画矩形
ctx.fillStyle = "#c02c38";
ctx.fillRect(20, 1, 120, 190);
ctx.fillRect(180, 1, 120, 300);
ctx.fillRect(340, 1, 120, 200);
ctx.fillRect(500, 1, 120, 400);
