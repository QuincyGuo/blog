    介绍

    # 介绍

    一个基于JavaScript的开源可视化图表库。

    # 基本柱形图

    # 引入库

    <script src="https://cdn.bootcdn.net/ajax/libs/echarts/5.1.2/echarts.js"></script>

    # 初始化图表容器

    <div id="main" style="width: 600px; height: 400px"></div>

    # 图表数据项配置

    // 基于DOM，初始化echarts实例
let dom = document.querySelector("div");
let myChart = echarts.init(dom);

// 指定图表的配置项与数据
var option = {
	// 1、图表标题
    title: {
      text: "ECharts 入门示例",
    },
    // 2、图表数据
    legend: {
      data: ["销量"],
    },
    // 3、x轴的配置
    xAxis: {
      data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"],
      show: false;	// 不显示横轴；
    },
    // 4、y轴的配置
    yAxis: {
      // 显示Y轴线及刻度
      axisLine: { show: true },
      axisTick: { show: true },
    },
    // 5、系列：图表类型及数据
    series: [
      {
        name: "销量",
        type: "bar",
        // bar柱状图 line折线图 pie饼图
        data: [5, 20, 36, 10, 10, 20],
      },
    ],
    // 6、图表的四边和画布四边的距离
    grid: {
      top: 0, left: 0, right: 0, bottom: 0,
    },
};

// 将配置项和数据应用于echarts实例
myChart.setOption(option);

    # 图表数据项配置

    var option1 = {
    title: {
      text: "ECharts 入门示例",
    },
    tooltip: {},
    xAxis: {
      data: ["衬衫", "羊毛衫", "雪纺衫", "裤子"],
    },
    yAxis: {
      axisLine: { show: true },
      axisTick: { show: true },

    series: [
      // 柱形图的数据项
      {
        name: "销量",
        type: "bar",
        data: [5, 20, 36, 10],
      },
      // 折线图
      {
        name: "销量",
        type: "line",
        data: [5, 20, 36, 10],
        // 折线图的线的颜色
        color: "#a61b29",
        // 折线图的区域填充颜色为渐变色
        areaStyle: {
            color: {
              type: 'linear', x: 0, y: 0, x2: 0, y2: 1,
              colorStops: [{
                offset: 0, color: 'red' // 0% 处的颜色
              }, {
                offset: 1, color: 'white' // 100% 处的颜色
              }],
            }
          }
      },
      // 饼状图
      {
        name: "销量",
        type: "pie",
        data: [
          { name: "衬衫", value: 5 },
          { name: "羊毛衫", value: 20 },
          { name: "雪纺衫", value: 36 },
          { name: "裤子", value: 10 },

    ],

    # 数据集dataset && 双坐标系

    数据集（dataset），将数据单独声明管理，而不是放到series系列里，可以被多个组件复用。更好地将逻辑和数据分离。

    双坐标系，一个图表内有多个x轴、y轴。

    # 定义数据集

    以二维数组的形式，定义数据集。每一行代表某一数据项在不同维度上的数据。

    let data = [
    // 每一列称为一个维度，分别是维度0,1,2
    ["衬衫", 20, 33],
    ["羊毛衫", 2, 23],
    ["雪纺衫", 30, 16],
    ["裤子", 40, 18],
];
// 比如，["衬衫", 20, 33]：
第一列，第一个维度，代表数据项名称；
第二列，第二个维度，代表销售量；
第三列，第三个维度，代表销售量增长率；

    # 使用数据集

    var option1 = {
    // 设置图表的数据集
    dataset: {
      source: data,
    },
    series: [
      {
        name: "柱形图",
        type: "bar",
        // 此处定义该图表使用数据集中哪个维度的数据，x轴：维度0；y轴：维度1；
        encode: {x: 0, y: 1 },
      },
      {
        name: "折线图",
        type: "line",
        encode: {x: 0, y: 2 },
      },
    ],
};

    # 双坐标系

    定义两个y轴

    // y轴的配置：两个Y轴坐标系
yAxis: [
  {
    // 显示Y轴线及刻度
    axisLine: { show: true },
    axisTick: { show: true },
  },
  {
    axisLine: { show: true },
    axisTick: { show: true },
  },
],

    让两个图表对应各自的y轴

    var option1 = {
    series: [
      {
        name: "柱形图",
        ...
        // 使用第一个Y轴坐标系
        yAxisIndex: 0,
      },
      // 折线图
      {
        name: "折线图",
        ...
        // 使用第二个Y轴坐标系
        yAxisIndex: 1,
      },
    ],
  };
